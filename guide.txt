# A guide to test the API using curl

# To get all users:
curl http://localhost:3000/

# To create a new user:
curl -X POST http://localhost:3000/ -H "Content-Type: application/json" -d '{"name":"John Doe"}'

# To update a user:
curl -X PUT http://localhost:3000/<user-id> -H "Content-Type: application/json" -d '{"name":"Jane Doe"}'

# To delete a user:
curl -X DELETE http://localhost:3000/<user-id>
